#include <WiFi.h>
#include <Firebase_ESP_Client.h>
#include <SPI.h>
#include <Adafruit_GFX.h>
#include <Adafruit_SSD1306.h>
#include <addons/TokenHelper.h>
#include <addons/RTDBHelper.h>

// Firebase
FirebaseData fbdo;
FirebaseAuth auth;
FirebaseConfig config;
FirebaseData stream;
FirebaseJson json;

// WiFi
#define WIFI_SSID ""
#define WIFI_PASSWORD ""

// Firebase
#define API_KEY ""
#define DATABASE_URL ""

// OLED
#define SCREEN_WIDTH 128
#define SCREEN_HEIGHT 64

#define OLED_MOSI 23
#define OLED_CLK 18
#define OLED_DC 2
#define OLED_CS 5
#define OLED_RESET 4

Adafruit_SSD1306 display(SCREEN_WIDTH, SCREEN_HEIGHT,
  OLED_MOSI, OLED_CLK, OLED_DC, OLED_RESET, OLED_CS);

// Pins
#define MIC_PIN 34
#define BUZZER_PIN 27
#define GREEN_LED 14
#define YELLOW_LED 12
#define RED_LED 13

// Thresholds
#define QUIET_THRESHOLD 1300
#define AVERAGE_THRESHOLD 2700
#define LOUD_THRESHOLD 4095

unsigned long lastUpdate = 0;
const unsigned long interval = 5000;
int maxSound = 0;

void setup() {

  Serial.begin(115200);

  pinMode(BUZZER_PIN, OUTPUT);
  pinMode(GREEN_LED, OUTPUT);
  pinMode(YELLOW_LED, OUTPUT);
  pinMode(RED_LED, OUTPUT);

  // OLED init
  if (!display.begin(SSD1306_SWITCHCAPVCC)) {
    Serial.println("OLED FAIL");
    for (;;);
  }

  display.clearDisplay();

  // WiFi
  WiFi.begin(WIFI_SSID, WIFI_PASSWORD);

  Serial.print("Connecting WiFi");

  while (WiFi.status() != WL_CONNECTED) {
    Serial.print(".");
    delay(300);
  }

  Serial.println("\nWiFi Connected!");

  // Firebase config
  config.api_key = API_KEY;
  config.database_url = DATABASE_URL;

  if (Firebase.signUp(&config, &auth, "", "")) {
      Serial.println("Firebase SignUp OK");
  } else {
      Serial.printf("SignUp Error: %s\n",
                    config.signer.signupError.message.c_str());
  }

  Firebase.begin(&config, &auth);
  Firebase.reconnectWiFi(true);

  int count = 0;

  while (!Firebase.ready() && count < 20) {
    Serial.print(".");
    delay(500);
    count++;
  }

  if (Firebase.ready()) {
    Serial.println("\nFirebase READY!");
  } else {
    Serial.println("\nFirebase FAILED!");
  }
}


void loop() {

  int soundValue = analogRead(MIC_PIN);
  if (soundValue > maxSound) {
    maxSound = soundValue;
  }

  if (millis() - lastUpdate >= interval) {
    lastUpdate = millis();

    String statusText;

    digitalWrite(GREEN_LED, LOW);
    digitalWrite(YELLOW_LED, LOW);
    digitalWrite(RED_LED, LOW);
    digitalWrite(BUZZER_PIN, LOW);

    if (maxSound < QUIET_THRESHOLD) {
      statusText = "Quiet";
      digitalWrite(GREEN_LED, HIGH);
    }
    else if (maxSound < AVERAGE_THRESHOLD) {
      statusText = "Average";
      digitalWrite(YELLOW_LED, HIGH);
    }
    else {
      statusText = "Loud";
      digitalWrite(RED_LED, HIGH);
      digitalWrite(BUZZER_PIN, HIGH);
    }

    Serial.print("Max Sound: ");
    Serial.print(maxSound);
    Serial.print(" | ");
    Serial.println(statusText);

    // OLED
    display.clearDisplay();
    display.setTextSize(1);
    display.setTextColor(WHITE);
    display.setCursor(10, 5);
    display.println("Noise Monitor");

    display.setCursor(10, 25);
    display.print("Level: ");
    display.println(maxSound);

    display.setCursor(10, 45);
    display.print("Status: ");
    display.println(statusText);

    display.display();

    // Firebase (ONLY ONCE - FIXED)
    if (!Firebase.RTDB.setInt(&fbdo, "NoiseMonitor/SoundLevel", maxSound)) {
      Serial.println(fbdo.errorReason());
    }

    if (!Firebase.RTDB.setString(&fbdo, "NoiseMonitor/Status", statusText)) {
      Serial.println(fbdo.errorReason());
    }

    maxSound = 0;
  }
}
